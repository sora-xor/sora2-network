pub trait CheckedMul: Sized + Mul<Self, Output = Self> {
    type Error;

    // Required methods

Expand description

Performs multiplication that returns None instead of wrapping around on underflow or overflow.

type Error

Required Methods§

source

fn checked_mul(&self, v: &Self) -> Option<Self>

Multiplies two numbers, checking for underflow or overflow. If underflow or overflow happens, None is returned.

source

fn cmul(&self, v: &Self) -> Result<Self, Self::Error>

Multiplies two numbers, checking for underflow or overflow. If underflow or overflow happens, Error is returned.
