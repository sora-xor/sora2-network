use bridge_types::{types::ChannelId, CHANNEL_INDEXING_PREFIX};
use codec::{Decode, Encode};

use jsonrpsee::{core::RpcResult as Result, proc_macros::rpc};
use sp_api::offchain::OffchainStorage;

pub use basic_channel::outbound::Message;
use sp_core::H256;

#[rpc(server, client)]
pub trait BasicChannelAPI {
    #[method(name = "basicChannel_commitment")]
    fn commitment(&self, commitment_hash: H256) -> Result<Option<Vec<Message>>>;
}

pub struct BasicChannelClient<S> {
    storage: S,
}

impl<S> BasicChannelClient<S> {
    /// Construct default `Template`.
    pub fn new(storage: S) -> Self {
        Self { storage }
    }
}

impl<S> BasicChannelAPIServer for BasicChannelClient<S>
where
    S: OffchainStorage + 'static,
{
    fn commitment(&self, commitment_hash: H256) -> Result<Option<Vec<Message>>> {
        let key = (CHANNEL_INDEXING_PREFIX, ChannelId::Basic, commitment_hash).encode();
        Ok(self
            .storage
            .get(sp_offchain::STORAGE_PREFIX, &key)
            .map(|value| Decode::decode(&mut &*value))
            .transpose()
            .map_err(|err| anyhow::Error::from(err))?)
    }
}
